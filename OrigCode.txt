signup php:
<?php
// Connect to the Database 
include('config.php');

if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $email = $_POST['email'];
    $pass = $_POST['pass'];
    
    // Check if the email already exists
    $check_query = "SELECT * FROM accounts WHERE email='$email'";
    $result = mysqli_query($conn, $check_query);
    if (mysqli_num_rows($result) > 0) {
        // Email already exists, show error message
        echo '<script>alert("Email already exists. Please try again."); window.location.href = "signup.html";</script>';
    } else {
        // Insert new user data into the database
        $insert_query = "INSERT INTO accounts (email, pass) VALUES ('$email', '$pass')";
        if (mysqli_query($conn, $insert_query)) {
            // New user created successfully, show success message
            echo '<script>alert("Account created successfully."); window.location.href = "signup.html";</script>';
        } else {
            // Error inserting user data
            echo '<script>alert("Error creating account. Please try again."); window.location.href = "signup.html";</script>';
        }
    }
}

mysqli_close($conn);
?>

        // Function to remember email and password if "Remember Me" is checked
        document.addEventListener('DOMContentLoaded', function() {
            if (localStorage.getItem('rememberEmail') && localStorage.getItem('rememberPass')) {
                emailInput.value = localStorage.getItem('rememberEmail');
                passwordInput.value = localStorage.getItem('rememberPass');
            }

            rememberMeCheckbox.addEventListener('change', function() {
                if (this.checked) {
                    localStorage.setItem('rememberEmail', emailInput.value);
                    localStorage.setItem('rememberPass', passwordInput.value);
                } else {
                    localStorage.removeItem('rememberEmail');
                    localStorage.removeItem('rememberPass');
                }
            });
        });

signin.php:
<?php
// Database connection
include('config.php');

// Start session
session_start();

if ($_SERVER["REQUEST_METHOD"] == "POST") {
    // Retrieve user input
    $email = $_POST['email'];
    $pass = $_POST['pass'];
    
    // Query to check if the email and password exist in the database
    $query = "SELECT * FROM accounts WHERE email='$email' AND pass='$pass'";
    $result = mysqli_query($conn, $query);
    
    // Check if any rows are returned
    if (mysqli_num_rows($result) > 0) {
        // User found, set session variables
        $_SESSION['email'] = $email;
        $_SESSION['loggedin'] = true;
        
        // Redirect to dashboard page
        header("Location: dashboard.html");
        exit();
    } else {
        // User not found, show error message
        echo '<script>alert("Invalid email or password. Please try again."); window.location.href = "signin.html";</script>';
    }
}

// Close database connection
mysqli_close($conn);
?>
